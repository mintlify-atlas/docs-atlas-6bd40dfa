## Overview

User types define the structure for case workers, authentication, user roles, and staff management in the Food For All DC system.

## CaseWorker

</ResponseField>

</ResponseField>

<ResponseField name="organization" type="string" required>

</ResponseField>

<ResponseField name="phone" type="string" required>

</ResponseField>

<ResponseField name="email" type="string" required>

Interface representing an authenticated user.

<ResponseField name="uid" type="string" required>

<ResponseField name="displayName" type="string | null">

<ResponseField name="user" type="AuthUser | null" required>

<ResponseField name="loading" type="boolean" required>

<ResponseField name="error" type="AuthError | null">

## Helper Function: canCreateUserType

Determines whether a user with a given role can create a new user with another role.

```typescript
export const canCreateUserType = (
  currentUserType: UserType,
  newUserType: UserType
): boolean => {
  switch (currentUserType) {
    case UserType.Admin:
      return [UserType.Admin, UserType.Manager, UserType.ClientIntake].includes(newUserType);
    case UserType.Manager:
      return newUserType === UserType.ClientIntake;
    default:
      return false;
  }
};
```

### Permission Rules

- **Admin** can create: Admin, Manager, ClientIntake
- **Manager** can create: ClientIntake only
- **ClientIntake** and **Driver** cannot create new users

## Usage Examples

### Working with UserType Enum

```typescript
import { UserType, canCreateUserType } from '@/types';

// Check if user can create another user type
const currentUser = UserType.Admin;
const canCreateAdmin = canCreateUserType(currentUser, UserType.Admin); // true
const canCreateManager = canCreateUserType(currentUser, UserType.Manager); // true

// Get all possible user types
const allUserTypes = Object.values(UserType);
console.log(allUserTypes); // ['Admin', 'Manager', 'ClientIntake', 'Driver']

// Check user role
function isAdmin(user: AuthUserRow): boolean {
  return user.role === UserType.Admin;
}

function hasManagerAccess(user: AuthUserRow): boolean {
  return user.role === UserType.Admin || user.role === UserType.Manager;
}
```

### Creating a Case Worker

```typescript
import { CaseWorker, ValidationErrors } from '@/types/user-types';

const newCaseWorker: Omit<CaseWorker, 'id'> = {
  name: 'Sarah Johnson',
  organization: 'DC Health Services',
  phone: '202-555-0123',
  email: 'sarah.johnson@dchealth.gov'
};

// Validate case worker data
function validateCaseWorker(
  caseWorker: Omit<CaseWorker, 'id'>
): ValidationErrors {
  const errors: ValidationErrors = {};
  
  if (!caseWorker.name || caseWorker.name.trim().length === 0) {
    errors.name = 'Name is required';
  }
  
  if (!caseWorker.organization || caseWorker.organization.trim().length === 0) {
    errors.organization = 'Organization is required';
  }
  
  const phoneRegex = /^\d{3}-\d{3}-\d{4}$/;
  if (!caseWorker.phone || !phoneRegex.test(caseWorker.phone)) {
    errors.phone = 'Phone must be in format: XXX-XXX-XXXX';
  }
  
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!caseWorker.email || !emailRegex.test(caseWorker.email)) {
    errors.email = 'Valid email is required';
  }
  
  return errors;
}

const errors = validateCaseWorker(newCaseWorker);
if (Object.keys(errors).length === 0) {
  console.log('Case worker is valid');
} else {
  console.error('Validation errors:', errors);
}
```

### Managing Authentication State

```typescript
import { AuthState, AuthUser, AuthError } from '@/types/user-types';
import { useState, useEffect } from 'react';

function useAuth() {
  const [authState, setAuthState] = useState<AuthState>({
    user: null,
    loading: true,
    error: null
  });
  
  useEffect(() => {
    // Subscribe to auth state changes
    const unsubscribe = onAuthStateChanged((user) => {
      if (user) {
        const authUser: AuthUser = {
          uid: user.uid,
          email: user.email,
          displayName: user.displayName,
          photoURL: user.photoURL,
          emailVerified: user.emailVerified,
          phoneNumber: user.phoneNumber,
          providerId: user.providerData[0]?.providerId
        };
        
        setAuthState({
          user: authUser,
          loading: false,
          error: null
        });
      } else {
        setAuthState({
          user: null,
          loading: false,
          error: null
        });
      }
    }, (error) => {
      const authError: AuthError = {
        code: error.code,
        message: error.message
      };
      
      setAuthState({
        user: null,
        loading: false,
        error: authError
      });
    });
    
    return () => unsubscribe();
  }, []);
  
  return authState;
}
```

### Role-Based Access Control

```typescript
import { UserType, canCreateUserType } from '@/types';

interface PermissionCheck {
  canManageClients: boolean;
  canManageDeliveries: boolean;
  canManageStaff: boolean;
  canViewReports: boolean;
}

function getUserPermissions(userType: UserType): PermissionCheck {
  switch (userType) {
    case UserType.Admin:
      return {
        canManageClients: true,
        canManageDeliveries: true,
        canManageStaff: true,
        canViewReports: true
      };
    
    case UserType.Manager:
      return {
        canManageClients: true,
        canManageDeliveries: true,
        canManageStaff: true,
        canViewReports: true
      };
    
    case UserType.ClientIntake:
      return {
        canManageClients: true,
        canManageDeliveries: false,
        canManageStaff: false,
        canViewReports: false
      };
    
    case UserType.Driver:
      return {
        canManageClients: false,
        canManageDeliveries: true,
        canManageStaff: false,
        canViewReports: false
      };
  }
}

// Usage
const permissions = getUserPermissions(UserType.ClientIntake);
if (permissions.canManageClients) {
  // Show client management UI
}
```

### Handling Authentication Errors

```typescript
import { AuthError } from '@/types/user-types';

function getAuthErrorMessage(error: AuthError): string {
  switch (error.code) {
    case 'auth/user-not-found':
      return 'No account found with this email address.';
    
    case 'auth/wrong-password':
      return 'Incorrect password. Please try again.';
    
    case 'auth/email-already-in-use':
      return 'An account with this email already exists.';
    
    case 'auth/weak-password':
      return 'Password should be at least 6 characters.';
    
    case 'auth/invalid-email':
      return 'Please enter a valid email address.';
    
    case 'auth/operation-not-allowed':
      return 'This sign-in method is not enabled.';
    
    case 'auth/too-many-requests':
      return 'Too many attempts. Please try again later.';
    
    default:
      return error.message || 'An error occurred during authentication.';
  }
}
```
