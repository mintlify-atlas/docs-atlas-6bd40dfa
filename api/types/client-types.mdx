## Overview

Client types define the structure for client profiles, dietary restrictions, and delivery details in the Food For All DC system.

### Checking Dietary Restrictions

```typescript
import { DietaryRestrictions } from '@/types/client-types';

function hasDietaryRestrictions(restrictions: DietaryRestrictions): boolean {
  return restrictions.lowSugar ||
         restrictions.kidneyFriendly ||
         restrictions.vegan ||
         restrictions.vegetarian ||
         restrictions.halal ||
         restrictions.lowSodium ||
         restrictions.heartFriendly ||
         restrictions.foodAllergens.length > 0 ||
         restrictions.other;
}

function getDietaryRestrictionsText(restrictions: DietaryRestrictions): string[] {
  const tags: string[] = [];
  
  if (restrictions.lowSugar) tags.push('Low Sugar');
  if (restrictions.kidneyFriendly) tags.push('Kidney Friendly');
  if (restrictions.vegan) tags.push('Vegan');
  if (restrictions.vegetarian) tags.push('Vegetarian');
  if (restrictions.halal) tags.push('Halal');
  if (restrictions.lowSodium) tags.push('Low Sodium');
  if (restrictions.heartFriendly) tags.push('Heart Friendly');
  
  restrictions.foodAllergens.forEach(allergen => {
    tags.push(`No ${allergen}`);
  });
  
  if (restrictions.other && restrictions.otherText) {
    tags.push(restrictions.otherText);
  }
  
  return tags;
}
```

### Updating Client Notes with Timestamp

```typescript
import { ClientProfile } from '@/types/client-types';

function updateClientNotes(client: ClientProfile, newNotes: string): ClientProfile {
  return {
    ...client,
    notes: newNotes,
    notesTimestamp: {
      notes: newNotes,
      timestamp: new Date()
    },
    updatedAt: new Date()
  };
}
```
