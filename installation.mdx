## Step 3: Configure Firebase

<Steps>
  <Step title="Create Firebase Project">
    1. Go to [Firebase Console](https://console.firebase.google.com/)
    2. Create a new project or use an existing one
    3. Enable Firestore Database
    4. Enable Firebase Authentication (Email/Password)
    5. Enable Cloud Functions
  </Step>
  
  <Step title="Get Firebase Configuration">
    In Firebase Console:
    1. Go to Project Settings
    2. Scroll to "Your apps" section
    3. Click "Add app" and select Web
    4. Copy the Firebase configuration object
  </Step>
  
  <Step title="Configure API Keys">
    Create or update `src/config/apiKeys.ts`:
    
    ```typescript src/config/apiKeys.ts
    export const firebaseConfig = {
      apiKey: "YOUR_API_KEY",
      authDomain: "your-project.firebaseapp.com",
      projectId: "your-project-id",
      storageBucket: "your-project.appspot.com",
      messagingSenderId: "123456789",
      appId: "1:123456789:web:abcdef"
    };
    
    // Google Maps API key for geocoding
    export const googleMapsApiKey = "YOUR_GOOGLE_MAPS_API_KEY";
    ```
    
    <Warning>
      Never commit API keys to version control. Add `apiKeys.ts` to `.gitignore`.
    </Warning>
  </Step>
  
  <Step title="Initialize Firebase Services">
    The app automatically initializes Firebase services in `src/auth/firebaseConfig.ts`:
    
    ```typescript src/auth/firebaseConfig.ts
    import { initializeApp, FirebaseApp } from "firebase/app";
    import { getFirestore, Firestore } from "firebase/firestore";
    import { getAuth, Auth } from "firebase/auth";
    import { getFunctions, Functions } from "firebase/functions";
    import { firebaseConfig } from "../config/apiKeys";
    
    const app: FirebaseApp = initializeApp(firebaseConfig);
    
    export const db = getFirestore(app);
    export const auth = getAuth(app);
    export const functions = getFunctions(app);
    
    // Enable device language for auth
    auth.useDeviceLanguage();
    ```
  </Step>
</Steps>

## Step 4: Set Up Firestore Collections

The app expects these Firestore collections (configured in `src/config/dataSources.ts`):

```typescript Collection Structure
- client-profile2        // Client profiles
- deliveries            // Delivery events
- drivers               // Driver information
- caseworkers           // Referral agency contacts
- tags                  // Client tags (single doc)
- daily-limits          // Delivery capacity by date
- users                 // App users and roles
```

<Note>
  Collections are created automatically when you add the first document. No manual setup required.
</Note>

## Step 5: Set Up Python Cloud Functions

<Steps>
  <Step title="Navigate to Functions Directory">
    ```bash
    cd functions-python
    ```
  </Step>
  
  <Step title="Create Virtual Environment">
    ```bash
    python3 -m venv venv
    source venv/bin/activate  # On Windows: venv\Scripts\activate
    ```
  </Step>
  
  <Step title="Install Python Dependencies">
    ```bash
    pip install -r requirements.txt
    ```
    
    Key Python packages:
    - `firebase-functions`: Cloud Functions SDK
    - `firebase-admin`: Admin SDK for Firestore
    - `scikit-learn`: Machine learning for clustering
    - `numpy`: Numerical computing

  </Step>
  
  <Step title="Return to App Directory">
    ```bash
    cd ..
    ```
  </Step>
</Steps>

## Step 7: Set Up Firebase Emulator (Optional)

<Warning>
  Use the Firebase Emulator for local testing only. Never use it with production data.
</Warning>

<Steps>
  <Step title="Login to Firebase">
    ```bash
    firebase login
    ```
  </Step>
  
  <Step title="Initialize Firebase in Project">
    ```bash
    firebase init
    ```
    
    Select:
    - Firestore
    - Functions
    - Emulators
    
    Choose Python for functions language.
  </Step>
  
  <Step title="Configure Emulators">
    Edit `firebase.json` to enable emulators:
    
    ```json firebase.json
    {
      "firestore": {
        "rules": "firestore.rules",
        "indexes": "firestore.indexes.json"
      },
      "functions": {
        "source": "functions-python"
      },
      "emulators": {
        "auth": {
          "port": 9099
        },
        "functions": {
          "port": 5001
        },
        "firestore": {
          "port": 8080
        },
        "ui": {
          "enabled": true,
          "port": 4000
        }
      }
    }
    ```
  </Step>
  
  <Step title="Start Emulators">
    ```bash
    firebase emulators:start
    ```
    
    Access the Emulator UI at [http://localhost:4000](http://localhost:4000)
  </Step>
  
  <Step title="Connect App to Emulator">
    Update `src/auth/firebaseConfig.ts` for development:
    
    ```typescript Development Mode
    import { connectFunctionsEmulator } from "firebase/functions";
    
    if (process.env.NODE_ENV === "development") {
      connectFunctionsEmulator(functions, "localhost", 5001);
    }
    ```
    
    This is already configured in the codebase.
  </Step>
</Steps>

## Step 8: Create Your First Admin User

<Steps>
  <Step title="Create User in Firebase Console">
    1. Go to Firebase Console > Authentication
    2. Click "Add user"
    3. Enter email and password
    4. Note the UID of the created user
  </Step>
  
  <Step title="Add User Document to Firestore">
    1. Go to Firestore Database
    2. Create a document in the `users` collection
    3. Set the document ID to the user's UID
    4. Add fields:
       ```json
       {
         "email": "admin@example.com",
         "role": "Admin",
         "name": "Admin User",
         "phone": ""
       }
       ```
  </Step>
  
  <Step title="Login to the App">
    Use the email and password to log in. You now have admin access.
  </Step>
</Steps>

## Key Files Explained

### Authentication

```typescript src/auth/ProtectedRoute.tsx
import { Navigate, Outlet } from "react-router-dom";
import { useAuth } from "./AuthProvider";
import { UserType } from "../types";

interface ProtectedRouteProps {
  allowedRoles: UserType[];
  redirectTo?: string;
}

const ProtectedRoute: React.FC<ProtectedRouteProps> = ({
  allowedRoles,
  redirectTo = "/",
  children,
}) => {
  const { userRole, loading } = useAuth();
  
  if (loading) {
    return <LoadingIndicator />;
  }
  
  if (!userRole || !allowedRoles.includes(userRole)) {
    return <Navigate to={redirectTo} replace />;
  }
  
  return children ? <>{children}</> : <Outlet />;
};
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Firebase connection errors">
    - Verify `apiKeys.ts` has correct Firebase config
    - Check Firebase project permissions
    - Ensure Firestore and Auth are enabled in Firebase Console
    - Try clearing browser cache and restarting dev server
  </Accordion>
  
  <Accordion title="npm install fails">
    - Delete `node_modules` and `package-lock.json`
    - Run `npm cache clean --force`
    - Try `npm install` again
    - Check Node.js version (must be 16+)
  </Accordion>
  
  <Accordion title="Python functions not working">
    - Verify virtual environment is activated
    - Check Python version (must be 3.11+)
    - Ensure all packages in `requirements.txt` are installed
    - Check Cloud Functions logs in Firebase Console
  </Accordion>
  
  <Accordion title="TypeScript errors">
    - Run `npm run lint` to see all errors
    - Check that all `@types/*` packages are installed
    - Verify `tsconfig.json` is correctly configured
    - Restart your IDE/editor
  </Accordion>
  
  <Accordion title="Map not loading">
    - Verify Google Maps API key is set in `apiKeys.ts`
    - Enable Maps JavaScript API in Google Cloud Console
    - Check browser console for errors
    - Ensure coordinates are valid [lat, lng] format
  </Accordion>
</AccordionGroup>
